function ns(name){
	var space,
		_super = this,
		i = 0,
		errors = [];
	name = (name.indexOf('.') &gt; -1)?name.split('.'):[name];
	for(;space = name[i++];){
		if(_super[space] !== undefined){
			var err = new Error(space + ' namespace already defined.');
			errors.push(err);
			_super = _super[space];
			continue;
		}
		_super[space] = new Namespace(space, _super);
		_super = _super[space];
	}
	if(errors.length &gt; 0){
		throw errors;
	}
	return _super;
}

when loading a function as a module the context of this will be the parent object, even if that is a function. watch out for chaining this as the context will become lost if you call this inside the parent function (this);

		throw new Error('Module already defined.');

	}

		var slice = Array.prototype.slice,
			res;
		if(!immutable){
			if(arguments.length === 1){
				set(arguments[0]);
			}			
		}
		res = (typeof module === 'object')?
				(module instanceof Array)?
						slice.call(module, 0): // copy the array

							Object.clone(module): // copy the object

								module; // return the primitive

			return get(res);
		}
		return res;
	}
	function changeSetter(fn){
		if(typeof fn === 'function'){
			set = function(value){
				var res = fn(value);
				defaultset(res);
			};
		}
	}
	function changeGetter(fn){
		if(typeof fn === 'function'){
			get = function(value){
				var res;
				if(value === undefined){
					return fn(res);
				}
				return value;
			};
		}
	}

	defaultget.changeSetter = changeSetter;

		this[name] = defaultget;
	}else{

function detectType(module){
	var type = 'Unknown';
	switch(typeof module){
	case 'string':
	case 'number':
	case 'boolean':
		type = 'Property';
		break;
	case 'object':
		if(module instanceof Array){
			type = 'Collection';
			break;
		}
		type = 'Singleton';
		break;
	case 'function':
		if(module.name &amp;&amp; module.name[0] === module.name[0].toUpperCase()){
			type = 'Constructor';
			break;
		}
		type = 'Method';
		break;
	default:
		break;		
	}
	return type;
}
